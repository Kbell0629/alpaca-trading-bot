# Cryptography for AES-GCM authenticated encryption of Alpaca credentials
# (supersedes the original stdlib-only HMAC stream cipher). REQUIRED for
# encrypting/decrypting Alpaca keys at rest. If this package fails to load
# at runtime, every saved credential becomes inaccessible — auth.py logs
# "[auth] CRITICAL: cryptography.AESGCM import failed: ..." to stderr and
# the dashboard returns 401 from /account on every refresh.
#
# Round-61 pt.13: tightened the upper bound from <46.0.0 to <44.0.0 after
# a Railway deploy went degraded with what was likely an ABI mismatch
# between Nix's Python 3.12 and a pre-built cryptography 45.x wheel
# (user-reported: "RuntimeError: cryptography package required for
# encrypt_secret()" surfaced via Save Paper Keys). Sticking to 42.x/43.x
# while we re-validate 44.x+ on the live image.
cryptography>=42.0.0,<44.0.0

# cffi — cryptography's transitive C-FFI dependency. Pinned EXPLICITLY so
# pip resolves it as a wheel even when the cryptography wheel doesn't
# pull it in correctly. Observed failure mode on Nixpacks Python 3.12:
# `from cryptography.hazmat.primitives.ciphers.aead import AESGCM` raises
# `ModuleNotFoundError: No module named '_cffi_backend'` at runtime,
# auth.py's bare except swallows it, _HAS_AESGCM=False, every saved
# credential becomes unreadable. Pinning cffi here forces pip to install
# the manylinux wheel which ships the _cffi_backend.so binary.
cffi>=1.16.0,<2.0.0

# zxcvbn — password strength estimator (Dropbox). Pure-Python, no ffi.
# Used in auth.create_user / change_password to reject weak passwords
# ("password", "abc12345", keyboard walks) that pass the 8-char min-length
# check but are trivially guessable. Graceful degradation at runtime: if
# this package fails to install, auth falls back to the 8-char minimum only.
zxcvbn>=4.4.28,<5.0.0

# yfinance — free Yahoo Finance scraper. Used by pead_strategy.py to
# fetch EPS actuals + estimates + surprise % needed for the
# Post-Earnings Announcement Drift strategy (Alpaca doesn't expose
# earnings-surprise data; yfinance has been the de-facto free source
# for ~5 years). Graceful degrade: if import fails at runtime, PEAD
# scoring returns 0 silently and other strategies are unaffected.
yfinance>=0.2.40,<0.3.0

# Round-11 expansion: sentry-sdk for error tracking. No-op at runtime
# unless SENTRY_DSN env var is set (paper users don't need it). Free
# tier covers 5K errors/month — more than enough for this bot's volume.
sentry-sdk>=2.0.0,<3.0.0

# Round-11 expansion: websocket-client for Alpaca real-time news stream
# (news_websocket.py). Module fails soft if the import is missing, so
# leaving this out is safe — just disables the live news feed.
websocket-client>=1.7.0,<2.0.0
