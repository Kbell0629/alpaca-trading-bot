# Cryptography for AES-GCM authenticated encryption of Alpaca credentials
# (supersedes the original stdlib-only HMAC stream cipher). Falls back to
# the legacy cipher at runtime if this package isn't installed, so removal
# doesn't brick existing deployments.
#
# Upper bound prevents auto-upgrade to a future major version with breaking
# API changes. Review and bump when cryptography releases 46.0.0; until then,
# Railway rebuilds pull a known-good version range.
cryptography>=42.0.0,<46.0.0

# zxcvbn — password strength estimator (Dropbox). Pure-Python, no ffi.
# Used in auth.create_user / change_password to reject weak passwords
# ("password", "abc12345", keyboard walks) that pass the 8-char min-length
# check but are trivially guessable. Graceful degradation at runtime: if
# this package fails to install, auth falls back to the 8-char minimum only.
zxcvbn>=4.4.28,<5.0.0

# yfinance — free Yahoo Finance scraper. Used by pead_strategy.py to
# fetch EPS actuals + estimates + surprise % needed for the
# Post-Earnings Announcement Drift strategy (Alpaca doesn't expose
# earnings-surprise data; yfinance has been the de-facto free source
# for ~5 years). Graceful degrade: if import fails at runtime, PEAD
# scoring returns 0 silently and other strategies are unaffected.
yfinance>=0.2.40,<0.3.0

# Round-11 expansion: sentry-sdk for error tracking. No-op at runtime
# unless SENTRY_DSN env var is set (paper users don't need it). Free
# tier covers 5K errors/month — more than enough for this bot's volume.
sentry-sdk>=2.0.0,<3.0.0

# Round-11 expansion: websocket-client for Alpaca real-time news stream
# (news_websocket.py). Module fails soft if the import is missing, so
# leaving this out is safe — just disables the live news feed.
websocket-client>=1.7.0,<2.0.0
